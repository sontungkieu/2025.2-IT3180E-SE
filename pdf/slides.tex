\begin{frame}{UC001 and UC002}
\begin{columns}[T]
\begin{column}{0.49\textwidth}
\textbf{UC001 Register and Verify Account}
\begin{itemize}
  \item Actor: Customer.
  \item Goal: register, verify contact/identity, then log in securely.
  \item Output: verified account and rental eligibility.
  \item Key exception: weak identity evidence, locked account, unpaid debt.
\end{itemize}
\end{column}
\begin{column}{0.49\textwidth}
\textbf{UC002 Manage Wallet and Rental Packages}
\begin{itemize}
  \item Actors: Customer, Payment Gateway.
  \item Goal: top up wallet or buy day/month package.
  \item Output: wallet transaction, balance, active package.
  \item Key exception: payment failed, package inactive.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{UC003 and UC004}
\begin{columns}[T]
\begin{column}{0.49\textwidth}
\textbf{UC003 Rent Bike by Hour}
\begin{itemize}
  \item Actors: Customer, GPS / Map Service.
  \item Goal: find bike, verify wallet/package and unlock bike.
  \item Output: active ride and bike in use.
  \item Key exception: insufficient wallet, active penalty, unlock failure.
\end{itemize}
\end{column}
\begin{column}{0.49\textwidth}
\textbf{UC004 End Ride and Settle Charges}
\begin{itemize}
  \item Actor: Customer.
  \item Goal: end ride and deduct final amount from wallet.
  \item Output: receipt, updated wallet, penalties if any.
  \item Key exception: wrong return area, unlocked bike, insufficient balance.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{UC005 Resolve Bike Maintenance Request}
\begin{itemize}
  \item Actors: Admin / Operator, Maintenance Staff.
  \item Customer is not a direct actor here; Customer only creates an issue report before this workflow.
  \item Admin / Operator: reviews report, marks bike unavailable, sets priority and assigns staff.
  \item Maintenance Staff: inspects, repairs, records result and closes/updates request.
  \item Output: maintenance request status and bike status.
\end{itemize}
\end{frame}

\begin{frame}{Rules and Penalties}
\begin{itemize}
  \item Rental is blocked if account is unverified, wallet is locked or unpaid penalty exists.
  \item Valid return requires correct station/area and locked bike.
  \item Package covers eligible riding time; overtime is charged from wallet.
  \item Penalty examples: wrong return area, failure to lock, damage/loss, overdue debt.
  \item Penalty records stay visible in ride receipt and wallet transaction history.
\end{itemize}
\end{frame}

\begin{frame}{Admin and Staff Overlap}
\begin{block}{Có giao nhau, nhưng không có Customer trong UC005}
Admin / Operator và Maintenance Staff cùng tham gia UC005 vì cả hai đều cần thiết để xử lý xe lỗi. Customer chỉ báo cáo sự cố, không trực tiếp thực hiện nghiệp vụ bảo trì.
\end{block}
\begin{itemize}
  \item Không tách các bước nhỏ như ``click assign'' hoặc ``update note'' thành use case.
  \item Các bước nhỏ được mô tả trong Basic Flow và Alternative Flows của UC005.
\end{itemize}
