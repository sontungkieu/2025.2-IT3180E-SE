\begin{frame}{Project Context and Rules}
\small
\begin{itemize}\tightitems
  \item Website quản lý thuê-trả xe đạp cho du khách và cư dân Ecopark.
  \item Quản lý điểm tập kết, loại xe, số lượng, ID xe duy nhất và trạng thái xe.
  \item Lần đầu thuê phải tạo tài khoản có email, CCCD/giấy tờ định danh, địa chỉ, số điện thoại.
  \item Nhận xe tại bãi với Station Staff / Manager: đặt cọc 200k + CCCD hoặc giấy tờ tùy thân.
  \item Giá thuê: 50k/1 giờ, 70k/2 giờ, 100k/3 giờ; trả muộn: 30k/30 phút.
  \item Cư dân có địa chỉ và thẻ cư dân hợp lệ được giảm 40\% giá thuê.
\end{itemize}
\end{frame}

\begin{frame}{Include / Extend Notes for UML}
\small
\begin{itemize}\tightitems
  \item Resident Customer generalizes to Customer/User; inherited associations do not need duplicate lines.
  \item UC001: Submit Resident Card \textit{extends} Create Customer Account.
  \item UC002: Filter Bicycle Type \textit{extends} View Station Bicycle Inventory; Send Rental Request \textit{includes} Select Available Bicycle.
  \item UC003: Confirm Bicycle Handover \textit{includes} Verify Identity Document and Collect Deposit; Cancel Request \textit{extends} Review Request.
  \item UC004: Apply Resident Discount and Apply Late Fee \textit{extend} Calculate Rental Fee; Issue Ticket \textit{includes} Calculate Rental Fee.
  \item UC005: Export Rental Reports \textit{extends} View Rental Statistics; Update Bicycle Status \textit{includes} Locate Bicycle.
\end{itemize}
\end{frame}

\begin{frame}{UC001 Main Flow: Register and Verify Customer Account}
\scriptsize
\textbf{Actors:} Customer/User, Resident Customer, Admin / Operator for manual review.
\begin{enumerate}\tightitems
  \item Customer/User chooses register or log in.
  \item System displays required account fields.
  \item Customer/User submits account and personal information.
  \item System validates format, duplicate email/phone and password.
  \item System creates account and stores customer profile.
  \item Customer/User submits CCCD/identity information.
  \item System marks rental eligibility when identity data is valid.
  \item Resident Customer optionally submits address and resident card.
  \item System verifies resident evidence or sends it to review.
  \item System displays account status, rental eligibility, discount and history.
\end{enumerate}
\end{frame}

\begin{frame}{UC001 Alternative Flows}
\scriptsize
\begin{enumerate}\tightitems
  \item Missing email, CCCD, phone or address: request completion.
  \item Duplicate email/phone: block registration and offer login/recovery.
  \item Weak password: ask for stronger password.
  \item Invalid/blocked identity number: reject rental eligibility.
  \item No resident card: allow normal customer rental, no 40\% discount.
  \item Resident card unreadable/mismatched: reject or pending discount.
  \item Suspicious resident profile: send to Admin / Operator review.
  \item Locked account: block rental request.
  \item Rental history unavailable: login still succeeds.
  \item Resident Customer inherits Customer/User and adds resident discount flow.
\end{enumerate}
\end{frame}

\begin{frame}{UC002 Main Flow: Find Station and Select Bicycle}
\scriptsize
\textbf{Actors:} Customer/User, GPS / Map Service.
\begin{enumerate}\tightitems
  \item Customer/User opens station search or rental screen.
  \item System asks for location or manual search area.
  \item GPS / Map Service returns location/search area.
  \item System lists nearest stations and available bicycle counts.
  \item Customer/User opens station detail.
  \item System shows bike IDs, types and current statuses.
  \item Customer/User filters type and selects an available bicycle.
  \item System re-checks account, station and bike availability.
  \item Customer/User confirms request and planned duration.
  \item System creates RentalRequest and pickup instruction.
\end{enumerate}
\end{frame}

\begin{frame}{UC002 Alternative Flows}
\scriptsize
\begin{enumerate}\tightitems
  \item Location permission denied: use manual station search.
  \item GPS / Map Service unavailable: use manual search.
  \item No nearby station: expand search range.
  \item Selected station inactive: suggest another station.
  \item Requested bike type unavailable: choose another type/station.
  \item No available bicycle: block request.
  \item Selected bicycle becomes rented/broken: refresh and reselect.
  \item Account lacks identity data: redirect to UC001.
  \item User cancels before confirmation: no RentalRequest.
  \item Request expires before pickup: cancel and release bicycle.
\end{enumerate}
\end{frame}

\begin{frame}{UC003 Main Flow: Process Bicycle Rental and Deposit}
\scriptsize
\textbf{Actors:} Customer/User, Station Staff / Manager.
\begin{enumerate}\tightitems
  \item Customer/User shows RentalRequest code at pickup station.
  \item Staff searches pending request.
  \item System displays customer, bike, station and planned duration.
  \item Staff checks CCCD/ID or accepted identity document.
  \item Staff confirms document matches customer account.
  \item Staff collects 200k deposit and records held document.
  \item System re-checks station and bicycle status.
  \item Staff hands over the correct bicycle.
  \item System creates Rental, deposit record and start time.
  \item System marks bicycle rented and shows handover receipt.
\end{enumerate}
\end{frame}

\begin{frame}{UC003 Alternative Flows}
\scriptsize
\begin{enumerate}\tightitems
  \item RentalRequest not found/expired: ask user to create new request.
  \item Request belongs to another station: block current processing.
  \item Customer forgot identity document: no handover.
  \item Document mismatch: reject and mark for review.
  \item Deposit below 200k: do not create rental.
  \item Customer refuses to leave document: do not hand over bicycle.
  \item Bicycle becomes broken/unavailable: block handover.
  \item Bicycle not at expected station: flag data mismatch.
  \item Customer wants another bike: return to UC002.
  \item Rental/deposit save fails: retry; do not mark bike rented.
  \item Receipt display/print fails: rental remains stored.
\end{enumerate}
\end{frame}

\begin{frame}{UC004 Main Flow: Return Bicycle and Issue Ticket}
\scriptsize
\textbf{Actors:} Customer/User, Station Staff / Manager.
\begin{enumerate}\tightitems
  \item Customer/User brings bicycle to any Ecopark station.
  \item Staff searches rental by rental ID, customer or bike ID.
  \item System displays rental, pickup station, start time and deposit.
  \item Staff inspects returned bicycle condition.
  \item Staff confirms return station and return time.
  \item System updates return station and actual duration.
  \item System calculates base fee: 50k/1h, 70k/2h, 100k/3h.
  \item System applies 40\% resident discount when eligible.
  \item System adds 30k/30min late fee when returned late.
  \item System issues ticket, stores history and updates bike status/location.
\end{enumerate}
\end{frame}

\begin{frame}{UC004 Alternative Flows}
\scriptsize
\begin{enumerate}\tightitems
  \item Return station differs from pickup: allow and update location.
  \item No in-progress rental found: manual operation review.
  \item Deposit mismatch: flag reconciliation.
  \item Bicycle damaged/needs repair: mark broken, not available.
  \item Staff entered wrong station/time: allow correction before ticket.
  \item Actual time below planned slot: apply configured price policy.
  \item Resident card not verified: no 40\% discount.
  \item Late time is partial 30 minutes: round by late-fee policy.
  \item Customer disputes late fee/time: issue ticket and flag review.
  \item Ticket cannot print/send: ticket remains stored.
  \item Status/location update fails: flag bike data for operations.
\end{enumerate}
\end{frame}

\begin{frame}{UC005 Main Flow: Manage Bikes, Stations and Rental Reports}
\scriptsize
\textbf{Actors:} Admin / Operator, Station Staff / Manager.
\begin{enumerate}\tightitems
  \item Actor opens operation management.
  \item Actor chooses station, bicycle, status, locate bike or reports.
  \item Station action: create/update name, address, coordinates, capacity, status.
  \item Bicycle action: create/update unique ID, type, current station and status.
  \item Status action: select bicycle and new operational state.
  \item System logs status/location change with actor and reason.
  \item Locate bike: search by bike ID or filters.
  \item Statistics: choose period, station, bike type or bike ID.
  \item System aggregates rentals, revenue, late fees, returns and statuses.
  \item System displays dashboard or exports report.
\end{enumerate}
\end{frame}

\begin{frame}{UC005 Alternative Flows}
\scriptsize
\begin{enumerate}\tightitems
  \item Staff lacks global permission: limit to assigned station.
  \item New station missing fields: reject save.
  \item Capacity below current bike count: reject update.
  \item Bike ID already exists: reject create/update.
  \item Rented bike manual station change: block until rental ends.
  \item Bike marked broken: hide from available list.
  \item Broken bike reopened: require inspection/repair note.
  \item Bike search has no result: show empty result.
  \item Report filters have no data: show empty report.
  \item Rentals pending reconciliation: mark statistics provisional.
  \item Report export fails: keep filters and allow retry.
\end{enumerate}
\end{frame}
