\begin{frame}{Mục tiêu bài toán}
\small
\begin{itemize}\tightitems
  \item Ecopark có nhiều bãi xe, nhiều loại xe và trạng thái xe thay đổi liên tục theo lượt thuê.
  \item Khách cần tìm bãi gần mình, chọn xe rảnh, gửi request và theo dõi lượt thuê.
  \item Staff/Admin cần kiểm giấy tờ, nhận cọc 200k, giao xe, nhận trả xe khác bãi và xuất vé.
  \item Quản lý cần dashboard doanh thu, trạng thái đội xe, audit log và CSV report theo bộ lọc.
  \item Nhóm triển khai một platform web có UI desktop/mobile và target deploy lên GCP.
\end{itemize}

\begin{frame}{API Và Backend Implementation}
\scriptsize
\begin{tabular}{p{0.24\textwidth}p{0.20\textwidth}p{0.46\textwidth}}
\toprule
\textbf{Nhóm API} & \textbf{Role} & \textbf{Endpoint chính} \\
\midrule
Auth/Profile & Public/Customer & \texttt{/api/auth/register}, \texttt{/api/auth/login}, \texttt{/api/customer/resident-card}. \\
Station/Bike & Customer & \texttt{/api/stations}, \texttt{/api/stations/:id/bikes}, \texttt{/api/rental-requests}. \\
Handover/Return & Staff/Admin & \texttt{/api/staff/pending-requests}, \texttt{/api/staff/handover}, \texttt{/api/staff/return}. \\
Admin/Report & Staff/Admin & \texttt{/api/admin/bikes}, \texttt{/api/admin/reports}, \texttt{/api/admin/bike-status-logs}. \\
\bottomrule
\end{tabular}
\vspace{0.25cm}
\begin{itemize}\tightitems
  \item Server-side validation cho họ tên, phone, CCCD/CMND và dữ liệu trùng.
  \item Rental/handover/return chạy qua transaction để tránh double-booking.
  \item Audit log lưu thay đổi trạng thái xe cùng người đổi, bãi và lý do.
\end{itemize}
\end{frame}

\begin{frame}{Dashboard Báo Cáo}
\small
\begin{itemize}\tightitems
  \item Revenue by period: tổng tiền vé theo ngày/tuần/tháng và filter station/bike.
  \item Fleet status distribution: số xe available/rented/broken để admin nắm sức khỏe đội xe.
  \item Rentals and late fees: lượt thuê và phụ thu trễ theo report rows.
  \item CSV export dùng đúng filter hiện tại, không xuất dữ liệu lệch màn hình đang xem.
  \item Audit panel giúp truy vết đổi trạng thái xe khi staff/admin thao tác.
\end{itemize}
\end{frame}

\begin{frame}{Testing Và Evidence}
\small
\begin{itemize}\tightitems
  \item Backend tests: đăng ký, duplicate data, held bikes, cancel, handover, return pricing, reports, audit log và concurrent sessions.
  \item UI smoke: desktop/mobile render, GSAP load, Three.js canvas pixel, Leaflet markers, no console errors, no horizontal overflow.
  \item UC smoke: GPS context, 2 customers + 2 admins, request/cancel, handover, return ticket, charts, audit log và CSV download.
  \item Report dùng screenshot thật từ app: auth, customer desktop/mobile, operations và \texttt{/gd}.
\end{itemize}
\end{frame}

\begin{frame}{Demo Script Khi Thuyết Trình}
\small
\begin{enumerate}\tightitems
  \item Mở platform và tab \texttt{/gd}; kéo xe gần bãi nhận và tua giờ nếu cần demo phụ thu.
  \item Customer đăng nhập, tìm bãi gần nhất, lọc loại xe, chọn xe và gửi request.
  \item Staff/Admin xác nhận giấy tờ, cọc 200k và giao xe.
  \item Staff nhận xe trả ở bãi khác, chọn tình trạng, xuất vé và xem tổng tiền.
  \item Admin xem dashboard chart, audit log và export CSV; nhắc target deploy GCP.
\end{enumerate}
\end{frame}
