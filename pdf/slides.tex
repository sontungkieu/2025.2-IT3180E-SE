\begin{frame}{Project Context and Rules}
\small
\begin{itemize}\tightitems
  \item Website quản lý thuê-trả xe đạp cho du khách và cư dân Ecopark.
  \item Quản lý điểm tập kết, loại xe, số lượng, ID xe duy nhất và trạng thái xe.
  \item Lần đầu thuê phải tạo tài khoản có email, CCCD/giấy tờ định danh, địa chỉ, số điện thoại.
  \item Nhận xe tại bãi với Station Staff / Manager: đặt cọc 200k + CCCD hoặc giấy tờ tùy thân.
  \item Giá thuê: 50k/1 giờ, 70k/2 giờ, 100k/3 giờ; trả muộn: 30k/30 phút.
  \item Cư dân có địa chỉ và thẻ cư dân hợp lệ được giảm 40\% giá thuê.
\end{itemize}
\end{frame}
