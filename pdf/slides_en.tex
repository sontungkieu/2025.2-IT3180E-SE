\begin{frame}{Project Context and Rules}
\small
\begin{itemize}\tightitems
  \item Website for bicycle rental and return services for Ecopark visitors and residents.
  \item Manage stations, bicycle types, quantities, globally unique bicycle IDs, and bicycle status.
  \item First-time renters must create an account with email, ID document, address, and phone number.
  \item Bicycle pickup is handled at the station with Station Staff / Manager: VND 200,000 deposit + ID document.
  \item Rental prices: VND 50k/1 hour, VND 70k/2 hours, VND 100k/3 hours; late return: VND 30k/30 minutes.
  \item Residents with a valid Ecopark address and resident card receive a 40\% rental discount.
\end{itemize}
\end{frame}

\begin{frame}{Actor Model}
\small
\begin{tabular}{p{0.30\textwidth}p{0.60\textwidth}}
\toprule
\textbf{Actor} & \textbf{Role} \\
\midrule
Customer/User & Regular renter: create account, find station, choose bicycle, pick up/return bicycle, view ticket/history. \\
Resident Customer & Specialized actor that inherits Customer/User; can submit resident card evidence for a 40\% discount. \\
Station Staff / Manager & Station-level operations: check ID/deposit, hand over and receive bicycles, update bicycles at the station; Manager supervises station scope. \\
Admin / Operator & Central/system-wide operations: manage stations/bicycles/policies, review evidence, view statistics; Admin has higher configuration rights than Operator. \\
GPS / Map Service & External service that supports finding the nearest bicycle station. \\
\bottomrule
\end{tabular}

\vspace{0.2cm}
\textbf{Not actors:} front-end, back-end, database, gate security.
\end{frame}

\begin{frame}{UC001 - 6. Basic Flow}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{enumerate}\tightitems
  \item Customer chooses account creation or login.
  \item System displays the required fields.
  \item Customer enters credentials and personal information.
  \item System validates formats, duplicate data, and password.
  \item System creates the account/profile.
\end{enumerate}
\column{0.48\textwidth}
\begin{enumerate}\setcounter{enumi}{5}\tightitems
  \item Customer adds ID document information.
  \item System marks the account as rental eligible.
  \item If resident, customer submits address and resident card.
  \item System verifies or moves the evidence to manual review.
  \item System shows account, discount, and history status.
\end{enumerate}
\end{columns}
\end{frame}

\begin{frame}{UC001 - 7. Alternative Flows}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}\tightitems
  \item Missing email/ID/phone/address: request completion.
  \item Email or phone already exists: block new account creation.
  \item Weak password: request re-entry.
  \item Invalid or blocked ID document: account is not rental eligible.
  \item Valid profile without resident proof: rent as regular customer.
\end{itemize}
\column{0.48\textwidth}
\begin{itemize}\tightitems
  \item No resident card or customer skips it: no discount.
  \item Resident card mismatches or cannot be read: rejected or pending.
  \item Suspicious profile: send to Admin / Operator review.
  \item Locked account: block rental request creation.
  \item Rental history temporarily fails: login still allowed.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{UC002 - 6. Basic Flow}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{enumerate}\tightitems
  \item Customer opens station search/rental feature.
  \item System asks for location or search area.
  \item GPS / Map Service returns location/area.
  \item System lists nearest stations and available counts.
  \item Customer selects a station to view details.
\end{enumerate}
\column{0.48\textwidth}
\begin{enumerate}\setcounter{enumi}{5}\tightitems
  \item System shows bicycles, type, bicycle ID, and status.
  \item Customer filters/selects an available bicycle.

  \item Customer confirms the rental request.
  \item System creates RentalRequest and pickup instructions.
\end{enumerate}
\end{columns}

\begin{frame}{UC002 - 7. Alternative Flows}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}\tightitems
  \item Location permission denied: enter location manually.
  \item GPS fails: use manual search.
  \item No nearby station: expand search radius.
  \item Station is suspended: suggest another station.
  \item Desired bicycle type is unavailable: change type or station.
\end{itemize}
\column{0.48\textwidth}
\begin{itemize}\tightitems
  \item No ready bicycle: do not create request.
  \item Bicycle is just rented/broken: refresh and select again.
  \item Account lacks identity data: return to UC001.
  \item Customer cancels before sending: no request is created.
  \item Request expires: auto-cancel and return bicycle to available.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{UC003 - 6. Basic Flow}

\begin{enumerate}\tightitems
  \item Customer presents the rental request code.
  \item Staff searches RentalRequest.

\begin{frame}{UC003 - 7. Alternative Flows}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}\tightitems
  \item Request not found/expired: create a new request.
  \item Request belongs to another station: do not process here.
  \item Customer forgot ID document: reject handover.
  \item Document mismatch: flag for review.
  \item Deposit below VND 200,000: do not create Rental.
  \item Customer refuses to leave document: reject handover.
\end{itemize}
\column{0.48\textwidth}
\begin{itemize}\tightitems
  \item Bicycle is broken/needs repair: block handover.
  \item Bicycle is not at station: record data mismatch.
  \item Customer wants another bicycle: return to UC002.
  \item Rental/Deposit save fails: retry, do not mark rented.
  \item Receipt cannot print/display: Rental remains stored.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{UC004 - 6. Basic Flow}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{enumerate}\tightitems
  \item Customer brings bicycle to an Ecopark station.
  \item Staff searches the Rental.
  \item System displays rental, bicycle, deposit, and time.
  \item Staff inspects bicycle condition.
  \item Staff confirms return station and return time.
\end{enumerate}
\column{0.48\textwidth}
\begin{enumerate}\setcounter{enumi}{5}\tightitems
  \item System updates return station and duration.
  \item System calculates base fee for 1/2/3-hour price.
  \item System applies 40\% discount if resident is verified.
  \item System adds VND 30k/30 minutes if returned late.
  \item System issues ticket, stores history, and updates bicycle.
\end{enumerate}
\end{columns}
\end{frame}

\begin{frame}{UC004 - 7. Alternative Flows}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}\tightitems
  \item Return station differs from pickup: accept and update location.
  \item Rental not found: send to manual review.
  \item Deposit mismatch: flag reconciliation.
  \item Bicycle broken/needs repair: mark broken, not rentable.
  \item Wrong station/time entered: allow correction before ticket.
  \item Actual duration below slot: follow pricing policy.
\end{itemize}
\column{0.48\textwidth}
\begin{itemize}\tightitems
  \item Not resident/not verified: no 40\% discount.
  \item Late time not exactly 30 minutes: round by policy.
  \item Late-fee dispute: store ticket and flag review.
  \item Ticket cannot print/send now: store for later view/print.
  \item Bicycle update fails: keep ticket and alert operations.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{UC005 - 6. Basic Flow}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{enumerate}\tightitems
  \item Actor opens operations management.
  \item Actor chooses station, bicycle, status, locate, or statistics.
  \item Station: create/update name, address, coordinates, capacity, status.

  \item Locate correct bicycle/location, then update actual status.

  \item System shows dashboard or exports report.
\end{enumerate}
\end{columns}

\begin{frame}{UC005 - 7. Alternative Flows}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}\tightitems
  \item Staff lacks global permission: limit to station scope.
  \item New station misses data: reject save.
  \item Capacity is below current bicycle count: reject reduction.
  \item Duplicate bicycle ID: block create/update.
  \item Bicycle is currently rented: block manual station change.
  \item Broken bicycle: remove from available list.
\end{itemize}
\column{0.48\textwidth}
\begin{itemize}\tightitems
  \item Reopen broken bicycle: require inspection/repair note.
  \item Bicycle ID not found: do not change data.
  \item Empty report filters: show empty report.
  \item Data pending reconciliation: mark statistics provisional.
  \item Report export fails: keep filters and allow retry.
\end{itemize}
\end{columns}
\end{frame}
